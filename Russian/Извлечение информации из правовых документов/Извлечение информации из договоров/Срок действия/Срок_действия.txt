// Поиск срока действия договора
rule: срок действия
{
    query: {phrase(3, действует or "срок действия", // Поиск ключевых слов
					
					optional("в течение"),
					
					{phrase(number(), год or месяц or неделя or день)}:длит // Поиск длительности действия договора
					or 
					phrase(до, {entity(Dates)}:оконч) // Поиск даты окончания договора
					
					)}:m

    result: Результат = title($m)
		attribute: Длительность = $длит
		attribute: Дата окончания = toentity(Dates, $оконч, field:=DateTime)

}
