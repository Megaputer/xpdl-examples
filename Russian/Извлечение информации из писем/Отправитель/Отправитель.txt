/* Правило для поиска отправителя письма */

/* Поиск всех компаний и организаций, обнаруженных стандартными правилами */
rule: поиск_компаний 
{
	query: {entity(Companies|Organizations)}:компания

	/* Первая упомянутая в начале документа компания считается отправителем */
	rule: поиск_отправителя
	{
		query: {follow(not $компания, position(20, $компания))}:отправитель
		
		result: Отправитель = $отправитель
	}
}
